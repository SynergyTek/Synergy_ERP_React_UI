const options = [
	{label: "This is an option"},
	{label: "This is another option"},
	{label: "This is a toast", onClick: () => {toast("This is a toast")}},
];

<ContextMenuTrigger options={options}>

	<div className={"h-32 w-56 outline-dotted rounded flex items-center justify-center"}>
		<Text color={"secondary"}>Right click here</Text>

</ContextMenuTrigger>
